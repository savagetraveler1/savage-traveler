    // --- SHARE + OPEN IN GMAPS ---
    function buildGMapsLink(){
      if (!(start && end)) return null;
      const coords = [start].concat(waypoints.map(w=>w.coord).filter(Boolean)).concat(poiWaypoints.map(p=>p.coord)).concat([end]);
      if (!coords.length) return null;
      return "https://www.google.com/maps/dir/" + coords.map(c=>c[1]+","+c[0]).join("/");
    }

    document.getElementById("openGMaps").addEventListener("click", ()=>{
      const url = buildGMapsLink();
      if (!url) { showToast("No route available",1500); return; }
      window.open(url,"_blank");
    });

    document.getElementById("shareLink").addEventListener("click", ()=>{
      const url = buildGMapsLink();
      if (!url) { showToast("No route available",1500); return; }
      navigator.clipboard.writeText(url).then(()=> showToast("Link copied!",1500));
    });
  </script>
